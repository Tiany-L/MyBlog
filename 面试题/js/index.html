# 先随便写个标题

# a == 1 && a == 2 && a == 3

  • 知识点
  • == 在判断是否相等时,如果数据类型不同时会进行类型转换。
  • 类型转换会调用 valueOf 和 toString
    • valueOf
    • toString 返回一个表示该对象的字符串，
    • window.toString === Object.prototype.toString
  • === 严格等于 不会进行类型转换
let a = {
  s: 1,
  toString: () => {
    return this.s++;
  },
};
if (a == 1 && a == 2 && a == 3) {
  console.log("成功1");
}

# 不规则数组找出数组的第二大值

//1
//对象的存储分为 常规属性和排序属性
//常规属性 一种不保证顺序的属性，创建对象会根据创建的顺序进行返回。
//排序属性 当使用正整数作为对象的key的时候，会按照从小到大的顺序进行排列
let arr = [1, 2, 3, 4, 23, 12, 434, 232, 212];
function findSecondLargest(arr) {
  let obj = {};
  arr.forEach((item) => {
    obj[item] = item;
  });
  return Object.keys(obj);
}
let sortArr = findSecondLargest(arr);
console.log(sortArr[sortArr.length - 2]);
